358 {
359  if (len <= str->alloc_len)
360  return 0;
361  if (!str->data && len <= sizeof(str->static_data))
362  str->alloc_len = len;
363  else if (str->alloc_len < len) {
364  char *new_data;
365  size_t new_len;
366  bool have_stack_data = !str->data;
367 
368  new_len = len + 4096;

371  return -ENOMEM;
372  // If we move beyond the on-stack buffer, copy the old data out

377  }

1097 static int utf8_to_utf32(const char *utf8, int len, uint32_t *utf32)
1098 {
1099  uint32_t ch = *utf8;
1100  uint8_t mask;
1101 
1102  if ((ch & 0x80) == 0) {
1103  len = 1;
1104  mask = 0x7f;
1105  } else if ((ch & 0xe0) == 0xc0 && len >= 2) {
1106  len = 2;
1107  mask = 0x1f;
1108  } else if ((ch & 0xf0) == 0xe0 && len >= 3) {
1109  len = 3;
1110  mask = 0xf;
1111  } else if ((ch & 0xf8) == 0xf0 && len >= 4) {
1112  len = 4;
1113  mask = 0x7;
1114  } else
1115  return -EINVAL;
1116 
1117  ch = 0;
1118  for (int i = 0; i < len; i++) {
1119  int shift = (len - i - 1) * 6;
1120  if (i == 0)
1121  ch |= ((uint32_t)(*utf8++) & mask) << shift;
1122  else
1123  ch |= ((uint32_t)(*utf8++) & 0x3f) << shift;
1124  }
1125 
1126  *utf32 = ch;
1127 
1128  return len;
1129 }

1131 static int pdf_add_text_spacing(struct pdf_doc *pdf, struct pdf_object *page,
1132  const char *text, float size, float xoff,
1133  float yoff, uint32_t colour, float spacing)
1134 {
1135  int ret;
1136  size_t len = text ? strlen(text) : 0;
1137  struct dstr str = INIT_DSTR;
1138  int alpha = (colour >> 24) >> 4;
1139 
1140  /* Don't bother adding empty/null strings */
1141  if (!len)
1142  return 0;
1143 
1144  dstr_append(&str, "BT ");
1145  dstr_printf(&str, "/GS%d gs ", alpha);
1146  dstr_printf(&str, "%f %f TD ", xoff, yoff);
1147  dstr_printf(&str, "/F%d %f Tf ", pdf->current_font->font.index, size);
1148  dstr_printf(&str, "%f %f %f rg ", PDF_RGB_R(colour), PDF_RGB_G(colour),
1149  PDF_RGB_B(colour));
1150  dstr_printf(&str, "%f Tc ", spacing);
1151  dstr_append(&str, "(");
1152 
1153  /* Escape magic characters properly */
1154  for (size_t i = 0; i < len;) {
1155  uint32_t code;
1156  int code_len;
1157  code_len = utf8_to_utf32(&text[i], len - i, &code);
1158  if (code_len < 0) {
1159  dstr_free(&str);
1160  return pdf_set_err(pdf, -EINVAL, "Invalid UTF-8 encoding");
1161  }
1162 
1163  if (code > 255) {
1164  /* We support *some* minimal UTF-8 characters */
1165  char buf[5] = {0};
1166  switch (code) {
1167  case 0x160:
1168  buf[0] = '\x8a';
1169  break;
1170  case 0x161:
1171  buf[0] = '\x9a';
1172  break;
1173  case 0x17d:
1174  buf[0] = '\x8e';
1175  break;
1176  case 0x17e:
1177  buf[0] = '\x9e';
1178  break;
1179  case 0x20ac:
1180  strcpy(buf, "\\200");
1181  break;
1182  default:
1183  dstr_free(&str);
1184  return pdf_set_err(pdf, -EINVAL,
1185  "Unsupported UTF-8 character: 0x%x 0o%o",
1186  code, code);
1187  }
1188  dstr_append(&str, buf);
1189  } else if (strchr("()\\", code)) {
1190  char buf[3];
1191  /* Escape some characters */
1192  buf[0] = '\\';
1193  buf[1] = code;
1194  buf[2] = '\0';
1195  dstr_append(&str, buf);
1196  } else if (strrchr("\n\r\t\b\f", code)) {
1197  /* Skip over these characters */
1198  ;
1199  } else {
1200  char buf[2];
1201  buf[0] = code;
1202  buf[1] = '\0';
1203  dstr_append(&str, buf);
1204  }
1205 
1206  i += code_len;
1207  }
1208  dstr_append(&str, ") Tj ");
1209  dstr_append(&str, "ET");
1210 
1211  ret = pdf_add_stream(pdf, page, dstr_data(&str));
1212  dstr_free(&str);
1213  return ret;
1214 }
